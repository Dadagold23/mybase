Options -Indexes
DirectoryIndex index.php index.html

<IfModule mod_rewrite.c>
    RewriteEngine On

    RewriteRule ^index\.php$ ./ [R=301,L,NC]
    RewriteRule ^about\.php$ about [R=301,L,NC]
    RewriteRule ^contact\.php$ contact [R=301,L,NC]

    RewriteCond %{REQUEST_FILENAME} -f [OR]
    RewriteCond %{REQUEST_FILENAME} -d
    RewriteRule ^ - [L]

    RewriteRule ^about/?$ about.php [L,QSA,NC]
    RewriteRule ^contact/?$ contact.php [L,QSA,NC]
</IfModule>

<FilesMatch "^\.">
    Require all denied
</FilesMatch>

<FilesMatch "\.(env|log|ini|sql|bak|sh|json|yml|yaml)$">
    Require all denied
</FilesMatch>

<IfModule mod_headers.c>
    Header set X-Frame-Options "SAMEORIGIN"
    Header set X-Content-Type-Options "nosniff"
    Header set Referrer-Policy "strict-origin-when-cross-origin"
    Header set Permissions-Policy "geolocation=(), microphone=(), camera=()"
</IfModule>
